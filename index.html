function assignCampaignExecutionTimes() { const SPREADSHEET_ID = '1BdzZj8Rr-9Twi7seNBGLomHm1FH__Rc-OtLEdTGyLmA'; const TARGET_GID = 834849830; const START_ROW = 968; const COL_START_DATE = 5; // E열 시작일 const COL_CHANNEL = 8; // H열 배너 여부 const COL_COUPON = 11; // K열 쿠폰 여부 const COL_STATUS = 18; // R열 상태 const COL_TIME = 19; // S열 실행시각 const COL_CAPA = 22; // V열 발송 모수 const DAILY_BANNER_CAPA = 2800000; const HOURLY_COUPON_CAPA = 120000; const BLOCKED_HOURS = [17, 18, 19]; const AVAILABLE_HOURS = []; for (let h = 1; h <= 23; h++) { if (!BLOCKED_HOURS.includes(h)) { AVAILABLE_HOURS.push(h); } } const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheets().find(s => s.getSheetId() === TARGET_GID); if (!sheet) { throw new Error('대상 GID의 시트를 찾을 수 없습니다.'); } const lastRow = sheet.getLastRow(); if (lastRow < START_ROW) return; const data = sheet.getRange(1, 1, lastRow, COL_CAPA).getValues(); const bannerUsedByDate = {}; const couponUsedByDateHour = {}; // 1. 이미 R열에 "일정 확정"인 캠페인 기준으로 기존 사용 CAPA 계산 for (let i = 1; i < data.length; i++) { const row = data[i]; const startDate = normalizeDate(row[COL_START_DATE - 1]); const channel = String(row[COL_CHANNEL - 1] || ''); const coupon = String(row[COL_COUPON - 1] || ''); const status = String(row[COL_STATUS - 1] || '').trim(); const execTime = row[COL_TIME - 1]; const capa = parseNumber(row[COL_CAPA - 1]); if (status !== '일정 확정') continue; if (!startDate || !capa) continue; if (channel.includes('배너')) { bannerUsedByDate[startDate] = (bannerUsedByDate[startDate] || 0) + capa; } if (coupon.includes('쿠폰') && execTime) { const hour = parseHour(execTime); if (hour !== null) { const key = `${startDate}_${hour}`; couponUsedByDateHour[key] = (couponUsedByDateHour[key] || 0) + capa; } } } const resultTimes = []; const resultStatuses = []; // 2. 968행 이후, R열에 "일정 확정"이 없는 캠페인만 신규 배정 for (let r = START_ROW; r <= lastRow; r++) { const row = data[r - 1]; const startDate = normalizeDate(row[COL_START_DATE - 1]); const channel = String(row[COL_CHANNEL - 1] || ''); const coupon = String(row[COL_COUPON - 1] || ''); const status = String(row[COL_STATUS - 1] || '').trim(); const currentTime = row[COL_TIME - 1]; const capa = parseNumber(row[COL_CAPA - 1]); if (status === '일정 확정') { resultTimes.push([currentTime]); resultStatuses.push([status]); continue; } if (!startDate || !capa) { resultTimes.push([currentTime]); resultStatuses.push(['배정 불가: 시작일 또는 V열 모수 없음']); continue; } if (channel.includes('배너')) { const usedBanner = bannerUsedByDate[startDate] || 0; if (usedBanner + capa > DAILY_BANNER_CAPA) { resultTimes.push([currentTime]); resultStatuses.push(['배정 불가: 배너 일 CAPA 초과']); continue; } bannerUsedByDate[startDate] = usedBanner + capa; } if (coupon.includes('쿠폰')) { const assignedHour = findAvailableCouponHour( startDate, capa, AVAILABLE_HOURS, couponUsedByDateHour, HOURLY_COUPON_CAPA ); if (assignedHour === null) { resultTimes.push([currentTime]); resultStatuses.push(['배정 불가: 쿠폰 시간 CAPA 부족']); continue; } const key = `${startDate}_${assignedHour}`; couponUsedByDateHour[key] = (couponUsedByDateHour[key] || 0) + capa; resultTimes.push([formatHour(assignedHour)]); resultStatuses.push(['일정 확정']); continue; } resultTimes.push(['01:00']); resultStatuses.push(['일정 확정']); } sheet.getRange(START_ROW, COL_TIME, resultTimes.length, 1).setValues(resultTimes); sheet.getRange(START_ROW, COL_STATUS, resultStatuses.length, 1).setValues(resultStatuses); SpreadsheetApp.getUi().alert('✅ 실행시각 자동배정이 완료되었습니다.'); } function findAvailableCouponHour(dateKey, capa, availableHours, couponUsedByDateHour, hourlyLimit) { for (const hour of availableHours) { const key = `${dateKey}_${hour}`; const used = couponUsedByDateHour[key] || 0; if (used + capa <= hourlyLimit) { return hour; } } return null; } function normalizeDate(value) { if (!value) return null; if (Object.prototype.toString.call(value) === '[object Date]') { return Utilities.formatDate(value, 'Asia/Seoul', 'yyyy-MM-dd'); } const parsed = new Date(value); if (!isNaN(parsed.getTime())) { return Utilities.formatDate(parsed, 'Asia/Seoul', 'yyyy-MM-dd'); } return String(value).trim(); } function parseHour(value) { if (!value) return null; if (Object.prototype.toString.call(value) === '[object Date]') { return Number(Utilities.formatDate(value, 'Asia/Seoul', 'H')); } const text = String(value).trim(); const match = text.match(/(\d{1,2})/); if (!match) return null; const hour = Number(match[1]); if (hour < 0 || hour > 23) return null; return hour; } function parseNumber(value) { if (typeof value === 'number') return value; const text = String(value || '').replace(/,/g, '').trim(); const num = Number(text); return isNaN(num) ? 0 : num; } function formatHour(hour) { return `${String(hour).padStart(2, '0')}:00`; } function onOpen() { SpreadsheetApp.getUi() .createMenu('🚀 캠페인 자동화') .addItem('실행시각 자동배정', 'assignCampaignExecutionTimes') .addToUi(); }
